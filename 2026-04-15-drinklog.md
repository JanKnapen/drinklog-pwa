# DrinkLog Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build DrinkLog, a SwiftUI iOS 17+ drink-logging app with four tabs (Home, Log, Manage, Data) backed by SwiftData.

**Architecture:** `TabView` with four `NavigationStack` tabs. `DrinkEntry` stores a nullifying `@Relationship` to `DrinkTemplate`. Template creation is deferred — "New" entries carry a `customName` until confirmed, at which point Confirm All creates the template and links it. `@Query` fetches drive all lists; no shared view model is needed.

**Tech Stack:** Swift 5.9, SwiftUI, SwiftData, Swift Charts, iOS 17+, Xcode 15+, xcodegen.

@Model
final class DrinkTemplate {
    var id: UUID
    var name: String
    var defaultMl: Double
    var defaultAbv: Double
    var usageCount: Int

    @Relationship(deleteRule: .nullify, inverse: \DrinkEntry.template)
    var entries: [DrinkEntry]

    init(name: String, defaultMl: Double, defaultAbv: Double, usageCount: Int = 0) {
        self.id = UUID()
        self.name = name
        self.defaultMl = defaultMl
        self.defaultAbv = defaultAbv
        self.usageCount = usageCount
        self.entries = []
    }
}

```swift
import Foundation
import SwiftData

@Model
final class DrinkEntry {
    var id: UUID
    var template: DrinkTemplate?
    /// Set only for unconfirmed "New" entries (pending template creation).
    /// Nil for "Enter ml" entries (no name) and confirmed entries (template is set).
    var customName: String?
    var ml: Double
    var abv: Double
    var timestamp: Date
    /// false = unconfirmed, true = confirmed
    var isMarked: Bool

    /// One standard drink = 10 ml pure alcohol
    var standardUnits: Double {
        (ml * abv / 100.0) / 10.0
    }

    /// The name to show in the UI. Nil for "Enter ml" entries.
    var displayName: String? {
        template?.name ?? customName
    }

    init(
        template: DrinkTemplate? = nil,
        customName: String? = nil,
        ml: Double,
        abv: Double,
        timestamp: Date = .now,
        isMarked: Bool = false
    ) {
        self.id = UUID()
        self.template = template
        self.customName = customName
        self.ml = ml
        self.abv = abv
        self.timestamp = timestamp
        self.isMarked = isMarked
    }
}
```

import SwiftUI
import SwiftData

struct NewDrinkSheet: View {
    @Environment(\.modelContext) private var context
    @Environment(\.dismiss) private var dismiss

    let existingTemplates: [DrinkTemplate]

    @State private var name = ""
    @State private var mlText = ""
    @State private var abvText = ""
    @State private var showDuplicateError = false

    private var trimmedName: String { name.trimmingCharacters(in: .whitespaces) }

    private var isDuplicate: Bool {
        existingTemplates.contains {
            $0.name.lowercased() == trimmedName.lowercased()
        }
    }

    private var isValid: Bool {
        !trimmedName.isEmpty && Double(mlText) != nil && Double(abvText) != nil
    }

    var body: some View {
        NavigationStack {
            Form {
                Section("Drink name") {
                    TextField("e.g. Lager, House Wine…", text: $name)
                        .autocorrectionDisabled()
                }
                Section("Amount") {
                    HStack {
                        TextField("ml", text: $mlText)
                            .keyboardType(.decimalPad)
                        Text("ml").foregroundStyle(.secondary)
                    }
                    HStack {
                        TextField("ABV", text: $abvText)
                            .keyboardType(.decimalPad)
                        Text("%").foregroundStyle(.secondary)
                    }
                }
                if let ml = Double(mlText), let abv = Double(abvText) {
                    Section {
                        let units = (ml * abv / 100.0) / 10.0
                        HStack {
                            Text("Standard units")
                            Spacer()
                            Text(units, format: .number.precision(.fractionLength(1)))
                                .foregroundStyle(.secondary)
                                .monospacedDigit()
                        }
                    }
                }
            }
            .navigationTitle("New Drink")
            .navigationBarTitleDisplayMode(.inline)
            .toolbar {
                ToolbarItem(placement: .cancellationAction) {
                    Button("Cancel") { dismiss() }
                }
                ToolbarItem(placement: .confirmationAction) {
                    Button("Log") { attemptSave() }
                        .disabled(!isValid)
                }
            }
            .overlay(alignment: .bottom) {
                if showDuplicateError {
                    ToastView(message: ""\(trimmedName)" already exists — use Other to log it")
                        .padding(.bottom, 20)
                        .transition(.move(edge: .bottom).combined(with: .opacity))
                }
            }
            .animation(.spring(response: 0.3), value: showDuplicateError)
        }
    }

    private func attemptSave() {
        guard !isDuplicate else {
            showDuplicateError = true
            DispatchQueue.main.asyncAfter(deadline: .now() + 3) { showDuplicateError = false }
            return
        }
        guard let ml = Double(mlText), let abv = Double(abvText) else { return }
        // Template creation is deferred until Confirm All runs.
        let entry = DrinkEntry(customName: trimmedName, ml: ml, abv: abv)
        context.insert(entry)
        dismiss()
    }
}

#Preview {
    NewDrinkSheet(existingTemplates: [])
        .modelContainer(for: [DrinkTemplate.self, DrinkEntry.self], inMemory: true)
}

import SwiftUI
import SwiftData

struct OtherDrinkSheet: View {
    @Environment(\.modelContext) private var context
    @Environment(\.dismiss) private var dismiss

    let templates: [DrinkTemplate]   // sorted by usageCount desc, from caller

    @State private var searchText = ""

    private var filtered: [DrinkTemplate] {
        guard !searchText.isEmpty else { return templates }
        return templates.filter { $0.name.localizedCaseInsensitiveContains(searchText) }
    }

    var body: some View {
        NavigationStack {
            List(filtered) { template in
                Button {
                    logTemplate(template)
                } label: {
                    HStack {
                        VStack(alignment: .leading, spacing: 2) {
                            Text(template.name)
                                .font(.body.weight(.medium))
                                .foregroundStyle(.primary)
                            Text("\(Int(template.defaultMl))ml · \(template.defaultAbv, specifier: "%.1f")% · \(template.standardUnits, specifier: "%.1f") units")
                                .font(.caption)
                                .foregroundStyle(.secondary)
                                .monospacedDigit()
                        }
                        Spacer()
                        Image(systemName: "plus.circle")
                            .foregroundStyle(.blue)
                    }
                    .contentShape(Rectangle())
                }
                .buttonStyle(.plain)
            }
            .listStyle(.insetGrouped)
            .searchable(text: $searchText, prompt: "Search drinks…")
            .navigationTitle("Other Drinks")
            .navigationBarTitleDisplayMode(.inline)
            .toolbar {
                ToolbarItem(placement: .cancellationAction) {
                    Button("Cancel") { dismiss() }
                }
            }
            .overlay {
                if filtered.isEmpty {
                    ContentUnavailableView.search(text: searchText)
                }
            }
        }
    }

    private func logTemplate(_ template: DrinkTemplate) {
        let entry = DrinkEntry(template: template, ml: template.defaultMl, abv: template.defaultAbv)
        context.insert(entry)
        template.usageCount += 1
        dismiss()
    }
}

import SwiftUI

struct HomeView: View {
    @Environment(\.modelContext) private var context
    @Query(sort: [SortDescriptor(\DrinkTemplate.usageCount, order: .reverse)])
    private var templates: [DrinkTemplate]

    @State private var showNewSheet = false
    @State private var showEnterMlSheet = false
    @State private var showOtherSheet = false

    private var topFive: [DrinkTemplate] { Array(templates.prefix(5)) }

    var body: some View {
        NavigationStack {
            ScrollView {
                VStack(alignment: .leading, spacing: 8) {
                    VStack(spacing: 8) {
                        ActionButton(
                            title: "New",
                            subtitle: "Create a new drink type",
                            systemImage: "plus.circle.fill"
                        ) { showNewSheet = true }

                        ActionButton(
                            title: "Enter ml",
                            subtitle: "Quick amount, no name",
                            systemImage: "drop.fill"
                        ) { showEnterMlSheet = true }

                        ActionButton(
                            title: "Other",
                            subtitle: "Pick from your drinks",
                            systemImage: "list.bullet"
                        ) { showOtherSheet = true }
                    }
                    .padding(.horizontal, 20)
                    .padding(.top, 8)

                    if !topFive.isEmpty {
                        FavoriteChipsRow(templates: topFive) { template in
                            logFromTemplate(template)

/// Pass template = nil to add a new template.

struct TemplateEditSheet: View {
    @Environment(\.modelContext) private var context
    @Environment(\.dismiss) private var dismiss

    let template: DrinkTemplate?
    let existingTemplates: [DrinkTemplate]

    @State private var name: String
    @State private var mlText: String
    @State private var abvText: String
    @State private var showDuplicateError = false

    private var trimmedName: String { name.trimmingCharacters(in: .whitespaces) }

    private var hasConfirmedEntries: Bool {
        template?.entries.contains { $0.isMarked } ?? false
    }

    private var isDuplicate: Bool {
        existingTemplates.contains {
            $0.id != template?.id &&
            $0.name.lowercased() == trimmedName.lowercased()
        }
    }

    private var isValid: Bool {
        let nameOk = !trimmedName.isEmpty
        if hasConfirmedEntries { return nameOk }
        return nameOk && Double(mlText) != nil && Double(abvText) != nil
    }

    init(template: DrinkTemplate? = nil, existingTemplates: [DrinkTemplate]) {
        self.template = template
        self.existingTemplates = existingTemplates
        _name = State(initialValue: template?.name ?? "")
        _mlText = State(initialValue: template.map { String(format: "%g", $0.defaultMl) } ?? "")
        _abvText = State(initialValue: template.map { String(format: "%g", $0.defaultAbv) } ?? "")
    }

    var body: some View {
        NavigationStack {
            Form {
                Section("Name") {
                    TextField("Drink name", text: $name)
                        .autocorrectionDisabled()
                }
                Section {
                    HStack {
                        TextField("ml", text: $mlText)
                            .keyboardType(.decimalPad)
                            .disabled(hasConfirmedEntries)
                        Text("ml").foregroundStyle(.secondary)
                    }
                    HStack {
                        TextField("ABV", text: $abvText)
                            .keyboardType(.decimalPad)
                            .disabled(hasConfirmedEntries)
                        Text("%").foregroundStyle(.secondary)
                    }
                } header: {
                    Text("Defaults")
                } footer: {
                    if hasConfirmedEntries {
                        Text("ml and ABV are locked because confirmed entries reference this template.")
                    }
                }
            }
            .navigationTitle(template == nil ? "New Template" : "Edit Template")
            .navigationBarTitleDisplayMode(.inline)
            .toolbar {
                ToolbarItem(placement: .cancellationAction) {
                    Button("Cancel") { dismiss() }
                }
                ToolbarItem(placement: .confirmationAction) {
                    Button("Save") { save() }.disabled(!isValid)
                }
            }
            .overlay(alignment: .bottom) {
                if showDuplicateError {
                    ToastView(message: ""\(trimmedName)" already exists")
                        .padding(.bottom, 20)
                        .transition(.move(edge: .bottom).combined(with: .opacity))
                }
            }
            .animation(.spring(response: 0.3), value: showDuplicateError)
        }
    }

    private func save() {
        guard !isDuplicate else {
            showDuplicateError = true
            DispatchQueue.main.asyncAfter(deadline: .now() + 3) { showDuplicateError = false }
            return
        }
        if let t = template {
            t.name = trimmedName
            if !hasConfirmedEntries {
                if let ml = Double(mlText), let abv = Double(abvText) {
                    t.defaultMl = ml
                    t.defaultAbv = abv
                }
            }
        } else {
            guard let ml = Double(mlText), let abv = Double(abvText) else { return }
            let newTemplate = DrinkTemplate(name: trimmedName, defaultMl: ml, defaultAbv: abv)
            context.insert(newTemplate)
        }
        dismiss()
    }
}

enum FilterPeriod: String, CaseIterable, Identifiable {
    case today = "Today"
    case week = "Week"
    case month = "Month"
    case threeMonths = "3M"
    case year = "Year"
    case all = "All"

    var id: String { rawValue }

    /// Returns the start date for this period, or nil for "All".
    func startDate(calendar: Calendar = .current, now: Date = .now) -> Date? {
        switch self {
        case .today:       return calendar.startOfDay(for: now)
        case .week:        return calendar.date(byAdding: .day,   value: -7,  to: now)
        case .month:       return calendar.date(byAdding: .month, value: -1,  to: now)
        case .threeMonths: return calendar.date(byAdding: .month, value: -3,  to: now)
        case .year:        return calendar.date(byAdding: .year,  value: -1,  to: now)

            }
        }
        .frame(maxWidth: .infinity, alignment: .leading)
        .padding(16)
        .background(Color(.secondarySystemBackground),
                    in: RoundedRectangle(cornerRadius: 12))
    }

## Design Principles Applied (frontend-design)

All views follow these constraints:

| Principle | Implementation |
|-----------|---------------|
| Typography hierarchy | `.font(.title2.weight(.bold))` for key numbers, `.caption.weight(.medium)` for labels, `.body.weight(.medium)` for list items |
| Numeric precision | `.monospacedDigit()` on all unit counts and numeric values |
| Colour restraint | `Color.blue` accent only on primary actions and active states; `Color(.secondarySystemBackground)` for cards |
| Press feedback | `PressableButtonStyle` with `.spring` scale on all tappable cards and chips |
| Motion | `.spring(response: 0.3, dampingFraction: 0.7)` on filter transitions; `.spring(response: 0.4)` on chart data changes; toast slide-up with `.spring` |
| Spacing | 20pt horizontal margins, 8pt gaps between buttons, 16pt card padding |
| Cards | `RoundedRectangle(cornerRadius: 12)` with `Color(.secondarySystemBackground)` — no explicit borders |
| Empty states | `ContentUnavailableView` throughout — no blank screens |
